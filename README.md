struct User {
    std::string status;
    std::string attitude;
    bool isFailed;

    User()
       :status("Trying"), attitude("Consistent"), isFailed(false) {}
};

int main(){
     std::vector<std::string> life = {"Challenge", "Failure", "Miserableness" };
     User user;

     while(user.status == "Trying"){
         if(user.attitude == "Consistent"){
              if(life.empty())
                 life.pop_back();
              else break;
         }
         else{
             user.isFailed = true;
             break;
         }
    }
}
